<?php
$conn = new mysqli("localhost", "root", "", "drnec");

if ($conn->connect_error) {
    die("Chyba připojení k databázi");
}

$conn->set_charset("utf8");

if ($_SERVER["REQUEST_METHOD"] == "POST") {
    $jmeno = trim($_POST["jmeno"]);
    $email = trim($_POST["email"]);
    $recenze = trim($_POST["recenze"]);

    if (!empty($jmeno) && !empty($email) && !empty($recenze)) {
        $stmt = $conn->prepare("INSERT INTO recenze (jmeno, email, recenze) VALUES (?, ?, ?)");
        $stmt->bind_param("sss", $jmeno, $email, $recenze);
        $stmt->execute();
        $stmt->close();

        echo "<p>Recenze byla úspěšně odeslána.</p>";
    }
}

$vysledek = $conn->query("SELECT * FROM recenze ORDER BY id DESC");

if ($vysledek->num_rows > 0) {
    while ($radek = $vysledek->fetch_assoc()) {
        echo "<div class='karticka'>";
        echo "<h3>" . htmlspecialchars($radek["jmeno"]) . "</h3>";
        echo "<small>" . htmlspecialchars($radek["email"]) . "</small>";
        echo "<p>" . htmlspecialchars($radek["recenze"]) . "</p>";
        echo "</div>";
    }
} else {
    echo "<p>Zatím tu nejsou žádné recenze.</p>";
}

$conn->close();
?>
